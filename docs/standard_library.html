Input Output (io)

            
using io

int _exec() {
    string name = io.input('What is your name? ')
    io.print('Hi there ' + name + '\n')
    int x = io.input_integer('Enter an integer.. ')
    x++
    io.print(x)
    return 0
}

Iterables (iter)

            
using io
using iter

int _exec() {
    list:int x = iter.range_int(0, 10, 1) // START END STEP
    for (int i in x) {
        io.print(i)
        io.print('\n')
    }
    return 0
}
